--시퀀스
drop sequence member_seq;
create sequence member_seq
start with 0
increment by 1
minvalue 0
nocycle;

--테이블
DROP TABLE member;
CREATE TABLE member (
	no      number       PRIMARY KEY, -- 번호
	id      varchar2(20) NOT NULL, -- 아이디
	pwd     varchar2(20) NOT NULL, -- 비밀번호
	name    varchar2(20) NOT NULL, -- 이름
	hp      varchar2(15) NULL,     -- 핸드폰
	hobby   varchar2(20) NULL,     -- 취미
	gender  varchar2(10) NOT NULL, -- 성별
	intro   CLOB         NULL,     -- 자기소개
	grade   varchar2(10) NOT NULL,  -- 학년
	regdate DATE 		 DEFAULT SYSDATE -- 등록일
);
